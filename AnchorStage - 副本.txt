const canvas = document.getElementById('game');
const cameraVideo = document.getElementById('cameraVideo');
const cameraStatus = document.getElementById('cameraStatus');
const ctx = canvas.getContext('2d');
const W = canvas.width, H = canvas.height;

// ---------- Level definition ----------
const GROUND_Y = 260;
const groundSegments = [{x1:0,x2:300},{x1:560,x2:W}];
const EDGE_X = 300;          // where the scripted jump triggers
const CHAR_START_X = 40;
const VX = 140;               // px/s constant horizontal speed
const GRAVITY = 850;
const JUMP_VELOCITY = -380;
const EXIT_X = W - 20;        // reaching here = success
const TIMELINE_DURATION = 6.5;

// ---------- State ----------
let time = 0;
let isPlaying = false;
let platform = null; // {x1,x2,tStart,tEnd}  y is always GROUND_Y
let lastDead = false, lastWin = false;
let deathTime = null, winTime = null;

let trailPoints = [];
const TRAIL_MAX = 22;
let lastHandLandmarks = null;
let handAnchors = [];
const HAND_CONNECTIONS = [
  [0,1],[1,2],[2,3],[3,4],
  [0,5],[5,6],[6,7],[7,8],
  [5,9],[9,10],[10,11],[11,12],
  [9,13],[13,14],[14,15],[15,16],
  [13,17],[17,18],[18,19],[19,20],
  [0,17]
];

// gesture state
let pinchActive = false;
let okActive = false;
let lastPinchActive = false;
let lastOkActive = false;
let pinchMode = null; // 'scrub' or 'draw'

// drawing state
let isDrawing = false, drawStart = null;
// scrub state
let isScrubbing = false;
// platform block drag state
let isDraggingPlatform = false, dragOffsetT = 0;

function getSupportY(x, t){
  for(const seg of groundSegments){
    if(x >= seg.x1 && x <= seg.x2) return GROUND_Y;
  }
  if(platform && t >= platform.tStart && t <= platform.tEnd && x >= platform.x1 && x <= platform.x2){
    return GROUND_Y;
  }
  return null;
}

// Pure replay function: derive character state at time t from scratch.
function computeStateAt(t){
  let st = { x:CHAR_START_X, y:GROUND_Y, vy:0, jumped:false, onSurface:true,
             dead:false, deathTime:null, success:false, successTime:null };
  const dt = 1/120;
  let cur = 0;
  while(cur < t){
    if(st.dead || st.success) break;
    const step = Math.min(dt, t - cur);
    cur += step;

    st.x += VX * step;

    if(!st.jumped && st.x >= EDGE_X){
      st.jumped = true;
      st.vy = JUMP_VELOCITY;
      st.onSurface = false;
    }
    if(!st.onSurface){
      st.vy += GRAVITY * step;
      st.y += st.vy * step;
    }

    const support = getSupportY(st.x, cur);
    if(support !== null && st.y >= support){
      st.y = support; st.vy = 0; st.onSurface = true;
    } else if (support === null) {
      st.onSurface = false;
    }

    if(st.y > H + 60){ st.dead = true; st.deathTime = cur; }
    if(st.x > EXIT_X){ st.success = true; st.successTime = cur; }
  }
  return st;
}

function distance(a, b){
  return Math.hypot(a.x - b.x, a.y - b.y);
}

function isPinch(landmarks){
  const pinchDistance = distance(landmarks[4], landmarks[8]);
  const handSize = distance(landmarks[0], landmarks[9]);
  return pinchDistance < Math.max(0.08, handSize * 0.22);
}

function isOkGesture(landmarks){
  const thumbIndex = distance(landmarks[4], landmarks[8]);
  const handSize = distance(landmarks[0], landmarks[9]);
  const openThreshold = Math.max(0.08, handSize * 0.22);
  const middleOpen = distance(landmarks[12], landmarks[9]) > openThreshold;
  const ringOpen = distance(landmarks[16], landmarks[13]) > openThreshold;
  const pinkyOpen = distance(landmarks[20], landmarks[17]) > openThreshold;
  return thumbIndex < Math.max(0.08, handSize * 0.18) && middleOpen && ringOpen && pinkyOpen;
}

function getHandCanvasPos(landmarks){
  const tip = landmarks[8];
  return {
    x: Math.min(W, Math.max(0, (1 - tip.x) * W)),
    y: Math.min(H, Math.max(0, tip.y * H))
  };
}

function finishHandDraw(){
  if(drawStart && drawCurrent){
    const dx = drawCurrent.x - drawStart.x;
    const dy = drawCurrent.y - drawStart.y;
    if(Math.abs(dx) > 40 && Math.abs(dx) > Math.abs(dy) * 1.2){
      const x1 = Math.max(0, Math.min(drawStart.x, drawCurrent.x));
      const x2 = Math.min(W, Math.max(drawStart.x, drawCurrent.x));
      const tStart = time;
      const tEnd = Math.min(TIMELINE_DURATION, tStart + 3.0);
      platform = { x1, x2, tStart, tEnd };
    }
  }
  isDrawing = false;
  drawStart = null;
}

function onHandsResults(results){
  if(!results.multiHandLandmarks || !results.multiHandLandmarks.length){
    if(lastPinchActive && pinchMode === 'draw') finishHandDraw();
    pinchActive = false;
    okActive = false;
    lastPinchActive = false;
    pinchMode = null;
    isScrubbing = false;
    cameraStatus.textContent = '未检测到手势';
    lastHandLandmarks = null;
    handAnchors = [];
    return;
  }

  const landmarks = results.multiHandLandmarks[0];
  console.log('onHandsResults landmarks:', landmarks);
  // store for rendering anchors
  lastHandLandmarks = landmarks;
  handAnchors = landmarks.map(lm => ({ x: Math.min(W, Math.max(0, (1 - lm.x) * W)), y: Math.min(H, Math.max(0, lm.y * H)) }));
  cameraStatus.textContent = `检测到手势（关键点: ${handAnchors.length}）`;
  pinchActive = isPinch(landmarks);
  okActive = isOkGesture(landmarks);
  const pos = getHandCanvasPos(landmarks);
  const isLowerHand = landmarks[8].y > 0.58;

  if(pinchActive){
    if(!lastPinchActive){
      if(isLowerHand){
        pinchMode = 'scrub';
        isScrubbing = true;
      } else {
        pinchMode = 'draw';
        drawStart = { x: pos.x, y: pos.y };
        drawCurrent = { x: pos.x, y: pos.y };
        isDrawing = true;
      }
    }

    if(pinchMode === 'scrub'){
      isPlaying = false;
      time = Math.min(Math.max(0, pos.x / W * TIMELINE_DURATION), TIMELINE_DURATION);
      cameraStatus.textContent = '捏住进度条，拖动时间';
    } else if(pinchMode === 'draw'){
      isPlaying = false;
      drawCurrent = { x: pos.x, y: pos.y };
      cameraStatus.textContent = '捏住画面，绘制平台';
    }
  } else {
    if(lastPinchActive && pinchMode === 'draw') finishHandDraw();
    isScrubbing = false;
    pinchMode = null;
  }

  if(okActive && !lastOkActive){
    isPlaying = true;
    cameraStatus.textContent = 'OK 手势，播放';
  }

  lastPinchActive = pinchActive;
  lastOkActive = okActive;
}

const hands = new Hands({
  locateFile: (file) => `https://cdn.jsdelivr.net/npm/@mediapipe/hands/${file}`
});
hands.setOptions({
  maxNumHands: 1,
  minDetectionConfidence: 0.5,
  minTrackingConfidence: 0.4,
  modelComplexity: 1,
  selfieMode: false
});
hands.onResults(onHandsResults);

const camera = new Camera(cameraVideo, {
  onFrame: async () => {
    await hands.send({ image: cameraVideo });
  },
  width: 640,
  height: 480
});
camera.start().then(() => {
  cameraStatus.textContent = '摄像头已就绪，开始手势控制';
}).catch((err) => {
  cameraStatus.textContent = '摄像头启动失败，请允许访问';
  console.error(err);
});

// ---------- Rendering ----------
function draw(){
  const st = computeStateAt(time);
  const cx = st.x;
  const cy = st.y - 14;

  trailPoints.push({ x: cx, y: cy });
  if(trailPoints.length > TRAIL_MAX) trailPoints.shift();

  ctx.clearRect(0,0,W,H);

  // sky
  const g = ctx.createLinearGradient(0,0,0,H);
  g.addColorStop(0,'#141826'); g.addColorStop(1,'#0E111A');
  ctx.fillStyle = g; ctx.fillRect(0,0,W,H);

  // ground segments
  ctx.fillStyle = '#1E2433';
  for(const seg of groundSegments){
    ctx.fillRect(seg.x1, GROUND_Y+6, seg.x2-seg.x1, H-GROUND_Y-6);
    ctx.fillStyle = '#2A3145';
    ctx.fillRect(seg.x1, GROUND_Y+6, seg.x2-seg.x1, 3);
    ctx.fillStyle = '#1E2433';
  }

  // gap void hint
  ctx.strokeStyle = 'rgba(255,92,92,0.15)';
  ctx.setLineDash([4,4]);
  ctx.beginPath();
  ctx.moveTo(300, GROUND_Y+6); ctx.lineTo(560, GROUND_Y+6);
  ctx.stroke();
  ctx.setLineDash([]);

  // platform
  if(platform){
    const active = time >= platform.tStart && time <= platform.tEnd;
    ctx.globalAlpha = active ? 1 : 0.28;
    ctx.fillStyle = active ? '#4FD1C5' : '#3A4356';
    ctx.fillRect(platform.x1, GROUND_Y+2, platform.x2-platform.x1, 6);
    if(active){
      ctx.shadowColor = '#4FD1C5'; ctx.shadowBlur = 12;
      ctx.fillRect(platform.x1, GROUND_Y+2, platform.x2-platform.x1, 6);
      ctx.shadowBlur = 0;
    }
    ctx.globalAlpha = 1;
  }

  // trail
  ctx.save();
  for(let i = 0; i < trailPoints.length; i++){
    const point = trailPoints[i];
    const alpha = (i + 1) / trailPoints.length * 0.35;
    const size = 12 - (trailPoints.length - i) * 0.24;
    ctx.fillStyle = `rgba(236,232,222,${alpha})`;
    ctx.beginPath();
    ctx.arc(point.x, point.y, Math.max(4, size), 0, Math.PI*2);
    ctx.fill();
  }
  ctx.restore();

  // in-progress drawing preview
  if(isDrawing && drawStart){
    ctx.strokeStyle = '#E8A33D';
    ctx.lineWidth = 3;
    ctx.beginPath();
    ctx.moveTo(drawStart.x, GROUND_Y+5);
    ctx.lineTo(drawCurrent.x, GROUND_Y+5);
    ctx.stroke();
  }

  // hand skeleton overlay
  if(handAnchors && handAnchors.length){
    ctx.save();
    // connections
    ctx.lineWidth = 2;
    for(const conn of HAND_CONNECTIONS){
      const a = handAnchors[conn[0]];
      const b = handAnchors[conn[1]];
      if(!a || !b) continue;
      ctx.strokeStyle = 'rgba(79,209,197,0.9)';
      ctx.beginPath();
      ctx.moveTo(a.x, a.y);
      ctx.lineTo(b.x, b.y);
      ctx.stroke();
    }
    // joints
    for(let i=0;i<handAnchors.length;i++){
      const p = handAnchors[i];
      ctx.fillStyle = i===8? 'rgba(232,163,61,0.95)' : 'rgba(236,232,222,0.9)';
      ctx.beginPath();
      ctx.arc(p.x, p.y, i===8?6:4, 0, Math.PI*2);
      ctx.fill();
    }
    ctx.restore();
  }

  // character
  ctx.save();
  ctx.translate(cx, cy);
  const wobble = st.onSurface ? Math.sin(time*10)*2 : 0;
  ctx.fillStyle = st.dead ? '#FF5C5C' : '#ECE8DE';
  ctx.shadowColor = st.dead ? '#FF5C5C' : 'rgba(236,232,222,.5)';
  ctx.shadowBlur = 10;
  ctx.beginPath();
  ctx.arc(0, wobble, 12, 0, Math.PI*2);
  ctx.fill();
  ctx.shadowBlur = 0;
  // simple legs
  ctx.strokeStyle = ctx.fillStyle;
  ctx.lineWidth = 3;
  ctx.beginPath();
  ctx.moveTo(-5, 10+wobble); ctx.lineTo(-8, 20+wobble);
  ctx.moveTo(5, 10+wobble); ctx.lineTo(8, 20+wobble);
  ctx.stroke();
  ctx.restore();

  updateUI(st);
}

// ---------- UI sync ----------
const playedBar = document.getElementById('playedBar');
const playhead = document.getElementById('playhead');
const mainTrack = document.getElementById('mainTrack');
const platformBlock = document.getElementById('platformBlock');
const failMsg = document.getElementById('failMsg');
const winMsg = document.getElementById('winMsg');
const timeReadout = document.getElementById('timeReadout');
const playBtn = document.getElementById('playBtn');

function updateUI(st){
  const pct = Math.min(time / TIMELINE_DURATION, 1) * 100;
  playedBar.style.width = pct + '%';
  playhead.style.left = pct + '%';
  timeReadout.textContent = time.toFixed(2) + 's / ' + TIMELINE_DURATION.toFixed(2) + 's';

  if(platform){
    platformBlock.style.display = 'block';
    const x1pct = (platform.tStart/TIMELINE_DURATION)*100;
    const x2pct = (platform.tEnd/TIMELINE_DURATION)*100;
    platformBlock.style.left = x1pct + '%';
    platformBlock.style.width = (x2pct-x1pct) + '%';
    const active = time >= platform.tStart && time <= platform.tEnd;
    platformBlock.classList.toggle('active', active);
  } else {
    platformBlock.style.display = 'none';
  }

  if(st.dead && !lastDead){
    deathTime = st.deathTime;
    isPlaying = false;
    failMsg.classList.add('show');
    setTimeout(()=>failMsg.classList.remove('show'), 1800);
  }
  if(!st.dead) { deathTime = null; }
  lastDead = st.dead;

  if(st.success && !lastWin){
    winTime = st.successTime;
    isPlaying = false;
    winMsg.classList.add('show');
    setTimeout(()=>winMsg.classList.remove('show'), 2200);
  }
  if(!st.success){ winTime = null; }
  lastWin = st.success;

  // death / win markers on timeline
  let deathEl = document.getElementById('deathMark');
  if(deathTime !== null){
    if(!deathEl){
      deathEl = document.createElement('div');
      deathEl.id = 'deathMark'; deathEl.className='death-mark'; deathEl.textContent='✕';
      mainTrack.appendChild(deathEl);
    }
    deathEl.style.left = (deathTime/TIMELINE_DURATION*100)+'%';
  } else if(deathEl){ deathEl.remove(); }

  let winEl = document.getElementById('winMark');
  if(winTime !== null){
    if(!winEl){
      winEl = document.createElement('div');
      winEl.id='winMark'; winEl.className='win-mark'; winEl.textContent='✓';
      mainTrack.appendChild(winEl);
    }
    winEl.style.left = (winTime/TIMELINE_DURATION*100)+'%';
  } else if(winEl){ winEl.remove(); }

  playBtn.textContent = isPlaying ? '⏸ 暂停' : '▶ 播放';
}

// ---------- Main loop ----------
let lastTs = null;
function loop(ts){
  if(lastTs === null) lastTs = ts;
  const dt = Math.min((ts-lastTs)/1000, 1/30);
  lastTs = ts;
  if(isPlaying){
    time += dt;
    if(time > TIMELINE_DURATION) time = TIMELINE_DURATION;
  }
  draw();
  requestAnimationFrame(loop);
}
requestAnimationFrame(loop);

// ---------- Canvas draw-platform interaction ----------
let drawCurrent = {x:0,y:0};
function canvasPos(e){
  const r = canvas.getBoundingClientRect();
  const scaleX = W / r.width, scaleY = H / r.height;
  return { x:(e.clientX-r.left)*scaleX, y:(e.clientY-r.top)*scaleY };
}
canvas.addEventListener('mousedown', (e)=>{
  const p = canvasPos(e);
  isDrawing = true; drawStart = p; drawCurrent = p;
  isPlaying = false;
});
window.addEventListener('mousemove', (e)=>{
  if(isDrawing){
    drawCurrent = canvasPos(e);
  }
});
window.addEventListener('mouseup', ()=>{
  if(isDrawing){
    const dx = drawCurrent.x - drawStart.x;
    const dy = drawCurrent.y - drawStart.y;
    if(Math.abs(dx) > 40 && Math.abs(dx) > Math.abs(dy)*1.2){
      const x1 = Math.max(0, Math.min(drawStart.x, drawCurrent.x));
      const x2 = Math.min(W, Math.max(drawStart.x, drawCurrent.x));
      const tStart = time;
      const tEnd = Math.min(TIMELINE_DURATION, tStart + 3.0);
      platform = { x1, x2, tStart, tEnd };
    }
    isDrawing = false; drawStart = null;
  }
});

// ---------- Timeline scrub interaction ----------
function seekFromEvent(e){
  const r = mainTrack.getBoundingClientRect();
  const pct = Math.min(1, Math.max(0, (e.clientX - r.left)/r.width));
  time = pct * TIMELINE_DURATION;
}
mainTrack.addEventListener('mousedown', (e)=>{
  isScrubbing = true; isPlaying = false; seekFromEvent(e);
});
window.addEventListener('mousemove', (e)=>{
  if(isScrubbing) seekFromEvent(e);
});
window.addEventListener('mouseup', ()=>{ isScrubbing = false; });

// ---------- Platform window drag ----------
const platformTrack = document.getElementById('platformTrack');
platformBlock.addEventListener('mousedown', (e)=>{
  e.stopPropagation();
  isDraggingPlatform = true;
  isPlaying = false;
  const r = platformTrack.getBoundingClientRect();
  const pct = (e.clientX - r.left)/r.width;
  dragOffsetT = pct*TIMELINE_DURATION - platform.tStart;
});
window.addEventListener('mousemove', (e)=>{
  if(isDraggingPlatform && platform){
    const r = platformTrack.getBoundingClientRect();
    const pct = (e.clientX - r.left)/r.width;
    const width = platform.tEnd - platform.tStart;
    let newStart = pct*TIMELINE_DURATION - dragOffsetT;
    newStart = Math.max(0, Math.min(TIMELINE_DURATION-width, newStart));
    platform.tStart = newStart;
    platform.tEnd = newStart + width;
  }
});
window.addEventListener('mouseup', ()=>{ isDraggingPlatform = false; });

// ---------- Buttons ----------
playBtn.addEventListener('click', ()=>{
  const st = computeStateAt(time);
  if(st.dead || (st.success && time>=TIMELINE_DURATION-0.01)) { time = 0; }
  isPlaying = !isPlaying;
});
document.getElementById('resetBtn').addEventListener('click', ()=>{
  time = 0; isPlaying = false;
  trailPoints = [];
});
document.getElementById('clearBtn').addEventListener('click', ()=>{
  platform = null;
  handAnchors = [];
  lastHandLandmarks = null;
});
